## Debug Commands for n8n CrashLoopBackOff

# ── APPLY PATCHES ─────────────────────────────────────────────────────────────
# Apply patch for VFZ2 (with scraper-creds)
kubectl patch deployment n8n -n preme-n8n --patch-file /Users/pulakmandal/context/ba/n8n/n8n-env-patch.yaml

# Apply patch for VFZ1 (without scraper-creds)
kubectl patch deployment n8n -n preme-n8n --patch-file /Users/pulakmandal/context/ba/n8n/n8n-env-patch-vfz1.yaml

# Remove envFrom from task-runner (VFZ1 — scraper-creds does not exist there)
kubectl patch deployment n8n -n preme-n8n --type=json -p='[{"op":"remove","path":"/spec/template/spec/containers/1/envFrom"}]'

# Restore original envFrom if accidentally removed (run before applying VFZ1 patch)
kubectl patch deployment n8n -n preme-n8n --type=json -p='[{"op":"add","path":"/spec/template/spec/containers/0/envFrom","value":[{"configMapRef":{"name":"n8n-app-config"}},{"secretRef":{"name":"n8n-app-secret"}}]}]'

# Check which container is crashing and restart counts
kubectl get pods -n preme-n8n

# Check liveness/readiness probe config and last state
kubectl describe pod -n preme-n8n -l app=n8n | grep -A10 "Liveness\|Readiness\|Last State\|Reason"

# Full pod description
kubectl describe pod -n preme-n8n -l app=n8n

# n8n container logs
kubectl logs deployment/n8n -n preme-n8n -c n8n --tail=50

# n8n container previous logs (if crashed)
kubectl logs deployment/n8n -n preme-n8n -c n8n --previous --tail=50

# task-runner container logs
kubectl logs deployment/n8n -n preme-n8n -c task-runner --tail=50

# task-runner previous logs
kubectl logs deployment/n8n -n preme-n8n -c task-runner --previous --tail=50

# Check all current env vars on deployment
kubectl set env deployment/n8n -n preme-n8n --list

# Get full deployment spec (useful to compare with working region)
kubectl get deployment n8n -n preme-n8n -o yaml

# Get only n8n container env vars
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.containers[0].env}' | tr ',' '\n'

# Get only n8n container envFrom (secrets/configmaps)
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.containers[0].envFrom}' | tr ',' '\n'

# Get volume mounts (check if any config files are mounted)
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.volumes}' | tr ',' '\n'

# Check if N8N_RUNNERS_AUTH_TOKEN exists on n8n container
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.containers[0].env}' | grep -i runners

# Check task-runner container env vars
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.containers[1].env}' | tr ',' '\n'

# Quick log check (last 20 lines per container)
kubectl logs deployment/n8n -n preme-n8n -c n8n --tail=20
kubectl logs deployment/n8n -n preme-n8n -c task-runner --tail=20

# Verify specific env vars are applied
kubectl set env deployment/n8n -n preme-n8n -c n8n --list | grep EXEC_ENV
kubectl set env deployment/n8n -n preme-n8n -c n8n --list | grep N8N_PORT
kubectl set env deployment/n8n -n preme-n8n -c n8n --list | grep RUNNERS_AUTH
