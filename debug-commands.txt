## Debug Commands for n8n CrashLoopBackOff

# ── APPLY PATCHES ─────────────────────────────────────────────────────────────
# Apply patch for VFZ2 (with scraper-creds)
kubectl patch deployment n8n -n preme-n8n --patch-file /Users/pulakmandal/context/ba/n8n/n8n-env-patch.yaml


# Restore original envFrom if accidentally removed (run before applying VFZ1 patch)
kubectl patch deployment n8n -n preme-n8n --type=json -p='[{"op":"add","path":"/spec/template/spec/containers/0/envFrom","value":[{"configMapRef":{"name":"n8n-app-config"}},{"secretRef":{"name":"n8n-app-secret"}}]}]'

# Check which container is crashing and restart counts
kubectl get pods -n preme-n8n

# Check liveness/readiness probe config and last state
kubectl describe pod -n preme-n8n -l app=n8n | grep -A10 "Liveness\|Readiness\|Last State\|Reason"

# Full pod description
kubectl describe pod -n preme-n8n -l app=n8n

# n8n container logs
kubectl logs deployment/n8n -n preme-n8n -c n8n --tail=50

# n8n container previous logs (if crashed)
kubectl logs deployment/n8n -n preme-n8n -c n8n --previous --tail=50

# task-runner container logs
kubectl logs deployment/n8n -n preme-n8n -c task-runner --tail=50

# task-runner previous logs
kubectl logs deployment/n8n -n preme-n8n -c task-runner --previous --tail=50

# Check all current env vars on deployment
kubectl set env deployment/n8n -n preme-n8n --list

# Get full deployment spec (useful to compare with working region)
kubectl get deployment n8n -n preme-n8n -o yaml

# Get only n8n container env vars
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.containers[0].env}' | tr ',' '\n'

# Get only n8n container envFrom (secrets/configmaps)
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.containers[0].envFrom}' | tr ',' '\n'

# Get volume mounts (check if any config files are mounted)
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.volumes}' | tr ',' '\n'

# Check if N8N_RUNNERS_AUTH_TOKEN exists on n8n container
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.containers[0].env}' | grep -i runners

# Check task-runner container env vars
kubectl get deployment n8n -n preme-n8n -o jsonpath='{.spec.template.spec.containers[1].env}' | tr ',' '\n'

# ── MODULE TROUBLESHOOTING ────────────────────────────────────────────────────
# Find where puppeteer-extra is installed in the n8n container
kubectl exec -it deployment/n8n -n preme-n8n -c n8n -- find / -name "puppeteer-extra" -type d 2>/dev/null | head -5

# Find where puppeteer-extra is installed in the task-runner container
kubectl exec -it deployment/n8n -n preme-n8n -c task-runner -- find / -name "puppeteer-extra" -type d 2>/dev/null | head -5

# Search all puppeteer-related packages
kubectl exec -it deployment/n8n -n preme-n8n -c n8n -- find /usr /home /app /opt -name "puppeteer*" -type d 2>/dev/null

# Find chromium browser
kubectl exec -it deployment/n8n -n preme-n8n -c n8n -- find /usr /home /app /opt -name "chromium*" 2>/dev/null

# Install puppeteer packages in task-runner using pnpm
kubectl exec -it deployment/n8n -n preme-n8n -c task-runner -- pnpm add puppeteer-extra puppeteer-extra-plugin-stealth puppeteer-core --prefix /tmp/pkgs

docker run --rm --entrypoint /usr/local/bin/node my-n8n:latest -e "require('puppeteer-extra'); console.log('OK')"

# Quick log check (last 20 lines per container)
kubectl logs deployment/n8n -n preme-n8n -c n8n --tail=20
kubectl logs deployment/n8n -n preme-n8n -c task-runner --tail=20

# Verify specific env vars are applied
kubectl set env deployment/n8n -n preme-n8n -c n8n --list | grep EXEC_ENV
kubectl set env deployment/n8n -n preme-n8n -c n8n --list | grep N8N_PORT
kubectl set env deployment/n8n -n preme-n8n -c n8n --list | grep RUNNERS_AUTH

# Allow external npm modules in Code nodes (required for puppeteer)
kubectl set env deployment/n8n -n preme-n8n \
  -c n8n NODE_FUNCTION_ALLOW_EXTERNAL=puppeteer-extra,puppeteer-extra-plugin-stealth,puppeteer-core \
  -c task-runner NODE_FUNCTION_ALLOW_EXTERNAL=puppeteer-extra,puppeteer-extra-plugin-stealth,puppeteer-core

# Allow ALL external modules (wildcard - use single quotes to prevent shell glob expansion)
kubectl set env deployment/n8n -n preme-n8n \
  -c n8n 'NODE_FUNCTION_ALLOW_EXTERNAL=*' \
  -c task-runner 'NODE_FUNCTION_ALLOW_EXTERNAL=*'

# Find exact variable name task-runner uses for module allowlist
kubectl exec -it deployment/n8n -n preme-n8n -c task-runner -- grep -n "disallowed\|allowedModules\|ALLOW_EXTERNAL" /opt/node_modules/@n8n/task-runner/javascript/task_runner.js | head -20

# task-runner js files are actually at /opt/runners/task-runner-javascript/dist/
# List config files
kubectl exec -it deployment/n8n -n preme-n8n -c task-runner -- ls /opt/runners/task-runner-javascript/dist/config/

# Read js-runner config (shows which env vars task-runner reads)
kubectl exec -it deployment/n8n -n preme-n8n -c task-runner -- cat /opt/runners/task-runner-javascript/dist/config/js-runner-config.js

# Find files that contain the "disallowed" module check logic
kubectl exec -it deployment/n8n -n preme-n8n -c task-runner -- grep -rl "disallowed" /opt/runners/task-runner-javascript/dist/ 2>/dev/null

# Read module resolver (key file for allowed module check logic)
kubectl exec -it deployment/n8n -n preme-n8n -c task-runner -- cat /opt/runners/task-runner-javascript/dist/js-task-runner/require-resolver.js

# Find where createRequireResolver is called (shows what allowedExternalModules n8n main sends)
kubectl exec -it deployment/n8n -n preme-n8n -c task-runner -- grep -rn "createRequireResolver\|allowedExternalModules" /opt/runners/task-runner-javascript/dist/ 2>/dev/null | grep -v ".map:"

# Bypass task-runner module check entirely
kubectl set env deployment/n8n -n preme-n8n -c task-runner N8N_RUNNERS_INSECURE_MODE=true

# Check configmap for conflicting NODE_FUNCTION values
kubectl get configmap n8n-app-config -n preme-n8n -o yaml | grep -i NODE_FUNCTION

# Verify env vars in running pod (not just deployment spec)
kubectl exec -it deployment/n8n -n preme-n8n -c n8n -- env | grep NODE_FUNCTION
kubectl exec -it deployment/n8n -n preme-n8n -c task-runner -- env | grep NODE_FUNCTION
