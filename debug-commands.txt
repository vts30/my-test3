# ── DOCKERFILE FIX FOR PUPPETEER IN TASK-RUNNER ───────────────────────────────
# Root cause: in N8N_RUNNERS_MODE=internal, n8n uses its OWN bundled task-runner
#   at /usr/local/lib/node_modules/n8n/node_modules/.pnpm/@n8n+task-runner.../
#   NOT the sidecar binary at /opt/runners/task-runner-javascript/
#
# Fixes that do NOT work:
#   - /opt/runners/task-runner-javascript/node_modules/ symlinks (wrong task-runner)
#   - /home/node/node_modules/ symlinks (not in pnpm resolution path)
#   - /usr/local/lib/node_modules/ symlinks (read-only filesystem)
#   - NODE_PATH (not inherited by internal task-runner subprocess)
#
# CORRECT FIX: dynamically find n8n's bundled task-runner and add symlinks there
# NOTE: @n8n/task-runner uses 'exports' to restrict subpaths — resolve main entry
#       and traverse up to find package root instead
# Add this RUN block in Dockerfile BEFORE the WORKDIR /home/node line (~line 97):
#
RUN node -e "\
const Module = require('module');\
const r = Module.createRequire('/usr/local/lib/node_modules/n8n/package.json');\
const fs = require('fs');\
const path = require('path');\
const mainPath = r.resolve('@n8n/task-runner');\
let pkgDir = path.dirname(mainPath);\
while (!fs.existsSync(path.join(pkgDir, 'package.json'))) { pkgDir = path.dirname(pkgDir); }\
const nodeModulesDir = path.join(pkgDir, 'node_modules');\
fs.mkdirSync(nodeModulesDir, {recursive: true});\
['puppeteer-extra','puppeteer-extra-plugin-stealth','puppeteer-core'].forEach(pkg => {\
  const link = path.join(nodeModulesDir, pkg);\
  if (!fs.existsSync(link)) fs.symlinkSync('/install/node_modules/' + pkg, link);\
});\
console.log('Symlinks created at', nodeModulesDir);\
"

# ── CREDENTIALS FIX: ADD TO N8N_ENVIRONMENT CONFIGMAP ─────────────────────────
# Root cause: process.env is EMPTY in Code nodes (internal task-runner subprocess
#   does not inherit container env vars). scraper-creds secret envFrom also does
#   not help. The only reliable way to pass custom vars to Code nodes is via
#   N8N_ENVIRONMENT configmap → accessible as $env.KEY in Code nodes.
#
# n8n UI Credentials (Settings > Credentials) do NOT work in Code nodes —
#   they are only accessible inside built-in node types via this.getCredentials().
#
# Fix: add TS_EMAIL, TS_PASSWORD, SCRAPER_PROXY to N8N_ENVIRONMENT JSON in configmap.
# After this: use $env.TS_EMAIL / $env.TS_PASSWORD / $env.SCRAPER_PROXY in node1.

# Step 1 — check current N8N_ENVIRONMENT value:
k get configmap n8n-app-config -n preme-n8n -o jsonpath='{.data.N8N_ENVIRONMENT}'

# Step 2 — export current value and merge in new keys:
# NOTE: N8N_ENVIRONMENT is stored as KEY=VALUE space-separated string, NOT JSON
k get configmap n8n-app-config -n preme-n8n -o jsonpath='{.data.N8N_ENVIRONMENT}' > /tmp/n8n_env.json
python3 << 'EOF'
with open('/tmp/n8n_env.json') as f:
    content = f.read().strip()

# Parse KEY=VALUE space-separated format
pairs = {}
for part in content.split():
    if '=' in part:
        k, v = part.split('=', 1)
        pairs[k] = v

# Add new entries
pairs['TS_EMAIL'] = 'your-tagesspiegel@email.de'
pairs['TS_PASSWORD'] = 'your-tagesspiegel-password'
pairs['SCRAPER_PROXY'] = 'http://PROXY_HOST:PORT'

# Serialize back to KEY=VALUE format
new_content = ' '.join(f'{k}={v}' for k, v in pairs.items())
print(new_content)

with open('/tmp/n8n_env_new.txt', 'w') as f:
    f.write(new_content)
EOF

# Step 3 — create patch JSON file (use temp file to avoid shell escaping issues):
python3 -c "
import json
new_val = open('/tmp/n8n_env_new.txt').read().strip()
print(json.dumps({'data': {'N8N_ENVIRONMENT': new_val}}))
" > /tmp/patch.json
cat /tmp/patch.json

# Step 4 — patch configmap and restart:
k patch configmap n8n-app-config -n preme-n8n --type merge -p "$(cat /tmp/patch.json)"
k rollout restart deployment/n8n -n preme-n8n
k rollout status deployment/n8n -n preme-n8n
