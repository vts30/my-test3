# ── DOCKERFILE FIX FOR PUPPETEER IN TASK-RUNNER ───────────────────────────────
# Root cause: in N8N_RUNNERS_MODE=internal, n8n uses its OWN bundled task-runner
#   at /usr/local/lib/node_modules/n8n/node_modules/.pnpm/@n8n+task-runner.../
#   NOT the sidecar binary at /opt/runners/task-runner-javascript/
#
# Fixes that do NOT work:
#   - /opt/runners/task-runner-javascript/node_modules/ symlinks (wrong task-runner)
#   - /home/node/node_modules/ symlinks (not in pnpm resolution path)
#   - /usr/local/lib/node_modules/ symlinks (read-only filesystem)
#   - NODE_PATH (not inherited by internal task-runner subprocess)
#
# CORRECT FIX: dynamically find n8n's bundled task-runner and add symlinks there
# NOTE: @n8n/task-runner uses 'exports' to restrict subpaths — resolve main entry
#       and traverse up to find package root instead
# Add this RUN block in Dockerfile BEFORE the WORKDIR /home/node line (~line 97):
#
RUN node -e "\
const Module = require('module');\
const r = Module.createRequire('/usr/local/lib/node_modules/n8n/package.json');\
const fs = require('fs');\
const path = require('path');\
const mainPath = r.resolve('@n8n/task-runner');\
let pkgDir = path.dirname(mainPath);\
while (!fs.existsSync(path.join(pkgDir, 'package.json'))) { pkgDir = path.dirname(pkgDir); }\
const nodeModulesDir = path.join(pkgDir, 'node_modules');\
fs.mkdirSync(nodeModulesDir, {recursive: true});\
['puppeteer-extra','puppeteer-extra-plugin-stealth','puppeteer-core'].forEach(pkg => {\
  const link = path.join(nodeModulesDir, pkg);\
  if (!fs.existsSync(link)) fs.symlinkSync('/install/node_modules/' + pkg, link);\
});\
console.log('Symlinks created at', nodeModulesDir);\
"
